import sys

def dfs(v):
    visited[v] = True
    dfs_list.append(v)
    for w in sorted(adj_list[v]):
        if not visited[w]:
            dfs(w)

def bfs(v):
    visited[v] = True
    queue = [v]
    while queue:
        v = queue.pop(0)
        bfs_list.append(v)
        for w in sorted(adj_list[v]):
            if not visited[w]:
                visited[w] = True
                queue.append(w)
